# MUST FOLLOW THIS SEQUENCE:
1.  ./scripts/dev_setup.sh                    # Install Rust, Flutter, Python

3.  cd core && cargo build --workspace       # Build Rust core
4.  python models/training/data_prep.py      # Prepare training data

11. ./scripts/build_android.sh               # Build for Android

13. cargo test --workspace                   # Final Rust validation
